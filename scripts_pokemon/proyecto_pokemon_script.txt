DROP TABLE pokemon;

/* CREATE TABLE tipo */
CREATE TABLE tipo
 (id_tipo INT AUTO_INCREMENT,
  nombre VARCHAR(20) NOT NULL,
  CONSTRAINT tipo_pk PRIMARY KEY (id_tipo));
 
 /* CREATE TABLE estado */
 CREATE TABLE estado
 (id_estado INT AUTO_INCREMENT,
  nombre VARCHAR(30) NOT NULL,
  CONSTRAINT estado_pk PRIMARY KEY (id_estado));
  
  /* CREATE TABLE movimiento */
  CREATE TABLE movimiento
 (id_movimiento INT AUTO_INCREMENT,
  nombre VARCHAR(30) NOT NULL,
  clase_movimiento VARCHAR(20) NOT NULL CHECK (clase_movimiento = 'ataque' OR clase_movimiento = 'estado' OR clase_movimiento = 'mejora'),
  categoria VARCHAR(20) NOT NULL CHECK (categoria =  'fisico' OR categoria = 'especial'),
  tipo INT,
  estamina INT ,
  potencia INT,
  estado INT,
  duracion INT,
  probabilidad INT,
  mejora VARCHAR(20),
  CONSTRAINT movimiento_pk PRIMARY KEY (id_movimiento),
  CONSTRAINT movimiento_tipo_fk FOREIGN KEY (tipo) REFERENCES tipo(id_tipo),
  CONSTRAINT movimiento_estado_fk FOREIGN KEY (estado) REFERENCES estado(id_estado));
  
  /* CREATE TABLE pokedex */
  CREATE TABLE pokedex
 (num_pokedex INT AUTO_INCREMENT,
  nombre VARCHAR(30) NOT NULL,
  nivel_evolucion INT,
  movimiento1 INT NOT NULL,
  movimiento2 INT,
  movimiento3 INT,
  movimiento4 INT,
  tipo INT,
  subtipo INT,
  img_front VARCHAR(200) NOT NULL,
  img_back VARCHAR(200) NOT NULL,
  CONSTRAINT pokedex_pk PRIMARY KEY (num_pokedex),
  CONSTRAINT pokedex_movimiento1_fk FOREIGN KEY (movimiento1) REFERENCES movimiento(id_movimiento),
  CONSTRAINT pokedex_movimiento2_fk FOREIGN KEY (movimiento2) REFERENCES movimiento(id_movimiento),
  CONSTRAINT pokedex_movimiento3_fk FOREIGN KEY (movimiento3) REFERENCES movimiento(id_movimiento),
  CONSTRAINT pokedex_movimiento4_fk FOREIGN KEY (movimiento4) REFERENCES movimiento(id_movimiento),
  CONSTRAINT pokedex_tipo_fk FOREIGN KEY (tipo) REFERENCES tipo(id_tipo),
  CONSTRAINT pokedex_subtipo_fk FOREIGN KEY (subtipo) REFERENCES tipo(id_tipo));
  
  /* CREATE TABLE entrenador */
  CREATE TABLE entrenador
 (id_entrenador INT AUTO_INCREMENT,
  nombre VARCHAR(40) NOT NULL UNIQUE,
  pokedollar INT NOT NULL,
  CONSTRAINT entrenador_pk PRIMARY KEY (id_entrenador));
  
  /* CREATE TABLE objeto */
  CREATE TABLE objeto
 (id_objeto INT AUTO_INCREMENT,
  nombre VARCHAR(30) NOT NULL,
  precio_compra INT,
  precio_venta INT,
  aumento INT,
  disminucion INT,
  descripcion VARCHAR(200),
  CONSTRAINT objeto_pk PRIMARY KEY (id_objeto));
  
  /* CREATE TABLE mochila */
  CREATE TABLE mochila
  (id_mochila INT AUTO_INCREMENT,
  id_entrenador INT,
  id_objeto INT,
  cantidad INT,
  CONSTRAINT mochila_pk PRIMARY KEY (id_mochila),
  CONSTRAINT mochila_objeto_fk FOREIGN KEY (id_objeto) REFERENCES objeto(id_objeto),
  CONSTRAINT mochila_entrenador_fk FOREIGN KEY (id_entrenador) REFERENCES entrenador(id_entrenador));
  
  /* CREATE TABLE equipo */
  CREATE TABLE equipo
  (id_equipo INT AUTO_INCREMENT,
   id_entrenador INT,
   CONSTRAINT equipo_pk PRIMARY KEY (id_equipo),
   CONSTRAINT equipo_entrenador_fk FOREIGN KEY (id_entrenador) REFERENCES entrenador(id_entrenador));
   
   /* CREATE TABLE caja */
   CREATE TABLE caja
  (id_caja INT AUTO_INCREMENT,
   id_entrenador INT,
   CONSTRAINT caja_pk PRIMARY KEY (id_caja),
   CONSTRAINT caja_entrenador_fk FOREIGN KEY (id_entrenador) REFERENCES entrenador(id_entrenador));
   
   /* CREATE TABLE pokemon */
   CREATE TABLE pokemon
   (id_pokemon INT AUTO_INCREMENT,
    num_pokedex INT NOT NULL,
    id_caja INT,
    id_entrenador INT,
    id_equipo INT,
    mote VARCHAR(30),
    sexo CHAR NOT NULL CHECK (sexo='m' OR sexo='h'),
    nivel INT NOT NULL DEFAULT 1,
    vitalidad INT NOT NULL,
    ataque INT NOT NULL,
    ataque_especial INT NOT NULL,
    defensa INT NOT NULL,
    defensa_especial INT NOT NULL,
    velocidad INT NOT NULL,
    estamina INT NOT NULL,
    fertilidad INT NOT NULL DEFAULT 5,
    objeto INT,
    movimiento1 INT NOT NULL,
    movimiento2 INT,
    movimiento3 INT,
    movimiento4 INT,
    estado INT,
    experiencia INT DEFAULT 0,
    CONSTRAINT pokemon_pk PRIMARY KEY (id_pokemon),
  CONSTRAINT pokemon_pokedex_fk FOREIGN KEY (num_pokedex) REFERENCES pokedex(num_pokedex),
  CONSTRAINT pokemon_caja_fk FOREIGN KEY (id_caja) REFERENCES caja(id_caja),
  CONSTRAINT pokemon_entrenador_fk FOREIGN KEY (id_entrenador) REFERENCES entrenador(id_entrenador),
  CONSTRAINT pokemon_equipo_fk FOREIGN KEY (id_equipo) REFERENCES equipo(id_equipo),
  CONSTRAINT pokemon_movimiento1_fk FOREIGN KEY (movimiento1) REFERENCES movimiento(id_movimiento),
  CONSTRAINT pokemon_movimiento2_fk FOREIGN KEY (movimiento2) REFERENCES movimiento(id_movimiento),
  CONSTRAINT pokemon_movimiento3_fk FOREIGN KEY (movimiento3) REFERENCES movimiento(id_movimiento),
  CONSTRAINT pokemon_movimiento4_fk FOREIGN KEY (movimiento4) REFERENCES movimiento(id_movimiento),
  CONSTRAINT pokemon_estado_fk FOREIGN KEY (estado) REFERENCES estado(id_estado));
